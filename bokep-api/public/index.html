Endpoints

GET /videos Mendapatkan daftar semua video
Parameter Type Description
pageoptional integer Nomor halaman (default: 1)
limitoptional integer Jumlah hasil per halaman (default: 30, max: 200)
qoptional string Kata kunci untuk pencarian
categoryoptional string Filter berdasarkan kategori
Response Example 
{
  "status": true,
  "total": 1209,
  "page": 1,
  "limit": 30,
  "totalPages": 41,
  "results": [
    {
      "id": "...",
      "title": "Bokep Indo - ...",
      "slug": "bokep-indo-...",
      "poster": "https://...",
      "categories": ["Bokep Indo", "..."],
      "sources_count": 3
    }
  ]
}
GET /videos/:slug Mendapatkan detail video termasuk source streaming
Parameter Type Description
slugrequired string Slug unik video
Response Example 
{
  "status": true,
  "data": {
    "id": "mcdowellforcongress-com",
    "title": "Bokep Indo - ...",
    "slug": "bokep-indo-...",
    "poster": "https://...",
    "categories": ["Bokep Indo"],
    "sources": [
      {
        "url": "https://...",
        "quality": "720p"
      }
    ]
  }
}
GET /videos?q={keyword} Mencari video berdasarkan judul
Parameter Type Description
qrequired string Kata kunci pencarian (min. 2 karakter)
limitoptional integer Maksimum hasil (default: 30)
Example Request 
GET /api/videos?q=indo&limit=10
GET /stats Mendapatkan statistik API
Response Example 
{
  "total": 1209,
  "categories": ["Bokep Indo", "Big Tits", ...],
  "lastUpdate": "2025-02-01T...",
  "source": "mcdowellforcongress.com"
}

 Contoh Penggunaan

            
            

                

                    JavaScript (Fetch)
                    
                

                
// Mendapatkan daftar video
const response = await fetch('https://bokep-api.vercel.app/api/videos?limit=10');
const data = await response.json();
console.log(data.results);

// Mencari video
const searchResponse = await fetch('https://bokep-api.vercel.app/api/videos?q=indo&limit=5');
const searchData = await searchResponse.json();
console.log(searchData.results);

// Mendapatkan detail video
const slug = 'bokep-indo-example-slug';
const detailResponse = await fetch(`https://bokep-api.vercel.app/api/videos/${slug}`);
const detailData = await detailResponse.json();
console.log(detailData.data.sources);

            

            
            

                

                    Python (Requests)
                    
                

                
import requests

# Base URL
BASE_URL = "https://bokep-api.vercel.app/api"

# Mendapatkan daftar video
response = requests.get(f"{BASE_URL}/videos", params={"limit": 10})
data = response.json()
print(data["results"])

# Mencari video
search_response = requests.get(f"{BASE_URL}/videos", params={"q": "indo", "limit": 5})
search_data = search_response.json()
print(search_data["results"])

# Mendapatkan detail video
slug = "bokep-indo-example-slug"
detail_response = requests.get(f"{BASE_URL}/videos/{slug}")
detail_data = detail_response.json()
print(detail_data["data"]["sources"])
